	\begin{frame}
		\frametitle{Objective and motivation}
		\begin{itemize}
		\item Research Question: Can next-day air pollution in Dublin be predicted from recent pollution levels, weather and traffic conditions, and what does this imply for DCC’s monitoring and intervention strategy?
		\item Target Variable of Interest:  Particulate matter pollution, specifically PM2.5, particles with a size (diameter) of less than 2.5 micrometers
		\item PM2.5 has been identified as the main contributor to poor health from air pollution as the very small particles can penetrate deep into the respiratory system, contributing to more pollution-related deaths than ozone (\href{https://ourworldindata.org/air-pollution-sources\#article-citation}{Our World in Data 2025})
		\item We aimed to create an accurate forecasting model that can predict next day PM2.5 levels, to support Dublin City Council in their monitoring and prevention of air pollution 
		(\href{https://www.dublincity.ie/environment-and-climate-action/controlling-air-pollution}{DCC Controlling Air Pollution})
		\end{itemize}
	\end{frame}

			\item Traffic Data: Traffic Volumes from SCATS Traffic Management System DCC

	\begin{frame}
		\frametitle{Discussion and recommendation}
		\begin{itemize}
			\item Final Model 
				\begin{itemize}
					\item Best full model: Ridge ARX
					\item Only a small gain over the simple \texttt{t-1} baseline\\ \texttt{RMSE 3.72 vs 3.87}
					\item Main signal: yesterday’s PM2.5; weather and traffic add only limited extra lift
				\end{itemize}
			\item Limitations
				\begin{itemize}
					\item Air View data is mobile, uneven, and weekday-only
					\item Dublin Airport weather used as city-wide weather proxy
					\item Traffic and weather are \textit{lagged by design}: predict tomorrow using what is known today
				\end{itemize}
			\item Recommendation to DCC 
				\begin{itemize}
					\item Use this as a light-touch next-day decision-support tool, not a standalone forecast
					\item Prioritise better data coverage and monitoring design over more complex algorithms
					\item Biggest gains likely from weekend coverage, stronger spatial balance, \textit{forecast} weather, and \textit{forward-looking} traffic inputs
				\end{itemize}
		\end{itemize}
